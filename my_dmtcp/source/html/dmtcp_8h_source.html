00138 /*
00139  * ComputationID
00140  *   ComputationID of a computation is the unique-pid of the first process of
00141  *   the computation. Even if that process dies, the rest of the computation
00142  *   retains the same computation ID.
00143  *
00144  *   With --enable-unique-checkpoint-filenames, the ComputationID also includes
00145  *   the checkpoint generation number (starting from 1). This number is same
00146  *   for the entire computation at a given point in time. Dmtcp coordinator
00147  *   increments this number prior to sending the SUSPEND message and is sent to
00148  *   the workers as a part of the SUSPEND message.

00203 #define NEXT_FNC(func)                                                      \
00204   ({                                                                        \
00205      static __typeof__(&func) _real_##func = (__typeof__(&func)) -1;        \
00206      if (_real_##func == (__typeof__(&func)) -1) {                          \
00207        __typeof__(&dlsym) dlsym_fnptr;                                      \
00208        if (dmtcp_prepare_wrappers) dmtcp_prepare_wrappers();                \
00209        dlsym_fnptr = (__typeof__(&dlsym)) dmtcp_get_libc_dlsym_addr();      \
00210        _real_##func = (__typeof__(&func)) (*dlsym_fnptr) (RTLD_NEXT, #func);\
00211      }                                                                      \
00212    _real_##func;})

00214 #define DMTCP_NEXT_EVENT_HOOK(event, data)                                  \
00215   do {                                                                      \
00216     static __typeof__(&dmtcp_event_hook) fn                                 \
00217       = (__typeof__(&dmtcp_event_hook)) -1;                                 \
00218     if ((void*) fn == (void*) -1) {                                         \
00219       fn = NEXT_FNC(dmtcp_event_hook);                                      \
00220     }                                                                       \
00221     if (fn != NULL) {                                                       \
00222       (*fn) (event, data);                                                  \
00223     }                                                                       \
00224   } while (0)
