00191 /* Code to initially initialize the thread pointer.  This might need
00192    special attention since 'errno' is not yet available and if the
00193    operation can cause a failure 'errno' must not be touched.  */
00194 # define TLS_INIT_TP(thrdescr, secondcall) \
00195   ({ void *_thrdescr = (thrdescr);                                            \
00196      tcbhead_t *_head = _thrdescr;                                            \
00197      union user_desc_init _segdescr;                                          \
00198      int _result;                                                             \
00199                                                                               \
00200      _head->tcb = _thrdescr;                                                  \
00201      /* For now the thread descriptor is at the same address.  */             \
00202      _head->self = _thrdescr;                                                 \
00203      /* New syscall handling support.  */                                     \
00204      INIT_SYSINFO;                                                            \
00205                                                                               \
00206      /* The 'entry_number' field.  Let the kernel pick a value.  */           \
00207      if (secondcall)                                                          \
00208        _segdescr.vals[0] = TLS_GET_GS () >> 3;                                \
00209      else                                                                     \
00210        _segdescr.vals[0] = -1;                                                \
00211      /* The 'base_addr' field.  Pointer to the TCB.  */                       \
00212      _segdescr.vals[1] = (unsigned long int) _thrdescr;                       \
00213      /* The 'limit' field.  We use 4GB which is 0xfffff pages.  */            \
00214      _segdescr.vals[2] = 0xfffff;                                             \
00215      /* Collapsed value of the bitfield:                                      \
00216           .seg_32bit = 1                                                      \
00217           .contents = 0                                                       \
00218           .read_exec_only = 0                                                 \
00219           .limit_in_pages = 1                                                 \
00220           .seg_not_present = 0                                                \
00221           .useable = 1 */                                                     \
00222      _segdescr.vals[3] = 0x51;                                                \
00223                                                                               \
00224      /* Install the TLS.  */                                                  \
00225      asm volatile (TLS_LOAD_EBX                                               \
00226                    "int $0x80\n\t"                                            \
00227                    TLS_LOAD_EBX                                               \
00228                    : "=a" (_result), "=m" (_segdescr.desc.entry_number)       \
00229                    : "0" (__NR_set_thread_area),                              \
00230                      TLS_EBX_ARG (&_segdescr.desc), "m" (_segdescr.desc));    \
00231                                                                               \
00232      if (_result == 0)                                                        \
00233        /* We know the index in the GDT, now load the segment register.        \
00234           The use of the GDT is described by the value 3 in the lower         \
00235           three bits of the segment descriptor value.                         \
00236                                                                               \
00237           Note that we have to do this even if the numeric value of           \
00238           the descriptor does not change.  Loading the segment register       \
00239           causes the segment information from the GDT to be loaded            \
00240           which is necessary since we have changed it.   */                   \
00241        TLS_SET_GS (_segdescr.desc.entry_number * 8 + 3);                      \
00242                                                                               \
00243      _result == 0 ? NULL                                                      \
00244      : "set_thread_area failed when setting up thread-local storage\n"; })

00271 /* Read member of the thread descriptor directly.  */
00272 # define THREAD_GETMEM(descr, member) \
00273   ({ __typeof (descr->member) __value;                                        \
00274      if (sizeof (__value) == 1)                                               \
00275        asm volatile ("movb %%gs:%P2,%b0"                                      \
00276                      : "=q" (__value)                                         \
00277                      : "0" (0), "i" (offsetof (struct pthread, member)));     \
00278      else if (sizeof (__value) == 4)                                          \
00279        asm volatile ("movl %%gs:%P1,%0"                                       \
00280                      : "=r" (__value)                                         \
00281                      : "i" (offsetof (struct pthread, member)));              \
00282      else                                                                     \
00283        {                                                                      \
00284          if (sizeof (__value) != 8)                                           \
00285            /* There should not be any value with a size other than 1,         \
00286               4 or 8.  */                                                     \
00287            abort ();                                                          \
00288                                                                               \
00289          asm volatile ("movl %%gs:%P1,%%eax\n\t"                              \
00290                        "movl %%gs:%P2,%%edx"                                  \
00291                        : "=A" (__value)                                       \
00292                        : "i" (offsetof (struct pthread, member)),             \
00293                          "i" (offsetof (struct pthread, member) + 4));        \
00294        }                                                                      \
00295      __value; })

00380 #define THREAD_ATOMIC_CMPXCHG_VAL(descr, member, newval, oldval) \
