00079 /*
00080  * struct MtcpRestartThreadArg
00081  *
00082  * DMTCP requires the virtualTids of the threads being created during
00083  *  the RESTARTING phase.  We use an MtcpRestartThreadArg structure to pass
00084  *  the virtualTid of the thread being created from MTCP to DMTCP.
00085  *
00086  * actual clone call: clone (fn, child_stack, flags, void *, ...)
00087  * new clone call   : clone (fn, child_stack, flags,
00088  *                           (struct MtcpRestartThreadArg *), ...)
00089  *
00090  * DMTCP automatically extracts arg from this structure and passes that
00091  * to the _real_clone call.
00092  *
00093  * NOTE: This structure will be moved to a more appropriate place once we have
00094  * finalized the code in threadlist.cpp.
00095  */
00096 struct MtcpRestartThreadArg {
00097   void * arg;
00098   pid_t virtualTid;
00099 };
