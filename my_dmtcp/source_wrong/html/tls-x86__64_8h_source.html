00129 /* Code to initially initialize the thread pointer.  This might need
00130    special attention since 'errno' is not yet available and if the
00131    operation can cause a failure 'errno' must not be touched.
00132 
00133    We have to make the syscall for both uses of the macro since the
00134    address might be (and probably is) different.  */
00135 # define TLS_INIT_TP(thrdescr, secondcall) \
00136   ({ void *_thrdescr = (thrdescr);                                            \
00137      tcbhead_t *_head = _thrdescr;                                            \
00138      int _result;                                                             \
00139                                                                               \
00140      _head->tcb = _thrdescr;                                                  \
00141      /* For now the thread descriptor is at the same address.  */             \
00142      _head->self = _thrdescr;                                                 \
00143                                                                               \
00144      /* It is a simple syscall to set the %fs value for the thread.  */       \
00145      asm volatile ("syscall"                                                  \
00146                    : "=a" (_result)                                           \
00147                    : "0" ((unsigned long int) __NR_arch_prctl),               \
00148                      "D" ((unsigned long int) ARCH_SET_FS),                   \

00178 /* Read member of the thread descriptor directly.  */
00179 # define THREAD_GETMEM(descr, member) \
00180   ({ __typeof (descr->member) __value;                                        \
00181      if (sizeof (__value) == 1)                                               \
00182        asm volatile ("movb %%fs:%P2,%b0"                                      \
00183                      : "=q" (__value)                                         \
00184                      : "0" (0), "i" (offsetof (struct pthread, member)));     \
00185      else if (sizeof (__value) == 4)                                          \
00186        asm volatile ("movl %%fs:%P1,%0"                                       \
00187                      : "=r" (__value)                                         \
00188                      : "i" (offsetof (struct pthread, member)));              \
00189      else                                                                     \
00190        {                                                                      \
00191          if (sizeof (__value) != 8)                                           \
00192            /* There should not be any value with a size other than 1,         \
00193               4 or 8.  */                                                     \
00194            abort ();                                                          \
00195                                                                               \
00196          asm volatile ("movq %%fs:%P1,%q0"                                    \
00197                        : "=r" (__value)                                       \
00198                        : "i" (offsetof (struct pthread, member)));            \
00199        }                                                                      \
00200      __value; })

00288 /* Atomic compare and exchange on TLS, returning old value.  */
00289 #define THREAD_ATOMIC_CMPXCHG_VAL(descr, member, newval, oldval) \
00290   ({ __typeof (descr->member) __ret;                                          \
00291      __typeof (oldval) __old = (oldval);                                      \
00292      if (sizeof (descr->member) == 4)                                         \
00293        asm volatile (LOCK_PREFIX "cmpxchgl %2, %%fs:%P3"                      \
00294                      : "=a" (__ret)                                           \
00295                      : "0" (__old), "r" (newval),                             \
00296                        "i" (offsetof (struct pthread, member)));              \
00297      else                                                                     \
00298        /* Not necessary for other sizes in the moment.  */                    \
00299        abort ();                                                              \
00300      __ret; })
